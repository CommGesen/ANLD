add_namespace = deluge_elections

# Radicals promise to nationalise industries and build a welfare state

country_event = {
	id = deluge_elections.1
	title = deluge_elections1title
	desc = deluge_elections1desc
	
	picture = GFX_report_event_election_vote
	
	is_triggered_only = yes

	trigger = {
		NOT = {
			OR = {
				tag = FRA
				tag = POL
				tag = TUR
				tag = USA
			}
		}
		NOT = { has_government = communism }
		NOT = { has_idea = welfare_state }
		OR = {
			communism > 0.5
			AND = {
				has_government = fascism
				fascism < 0.5
				communism > 0.4
				democratic < 0.5
				neutrality < 0.5
			}
			AND = {
				has_government = fascism
				fascism < 0.4
				communism > 0.3
				democratic < 0.4
				neutrality < 0.4
			}
			AND = {
				has_government = fascism
				fascism < 0.3
				communism > 0.2
				democratic < 0.3
				neutrality < 0.3
			}
			AND = {
				has_government = democratic
				democratic < 0.5
				communism > 0.4
				fascism < 0.5
				neutrality < 0.5
			}
			AND = {
				has_government = democratic
				democratic < 0.4
				communism > 0.3
				fascism < 0.4
				neutrality < 0.4
			}
			AND = {
				has_government = democratic
				democratic < 0.3
				communism > 0.2
				fascism < 0.3
				neutrality < 0.3
			}
			AND = {
				has_government = neutrality
				neutrality < 0.5
				communism > 0.4
				fascism < 0.5
				democratic < 0.5
			}
			AND = {
				has_government = neutrality
				neutrality < 0.4
				communism > 0.3
				fascism < 0.4
				democratic < 0.4
			}
			AND = {
				has_government = neutrality
				neutrality < 0.3
				communism > 0.2
				fascism < 0.3
				democratic < 0.3
			}
		}
	}
	
	option = { # Let them win
		name = deluge_let_them_win
	
		add_popularity = {
			ideology = communism
			popularity = 0.05
		}
		
		set_politics = {
			ruling_party = communism
			elections_allowed = yes
		}
		
		add_ideas = welfare_state
		
		ai_chance = {
			factor = 1
			
			modifier = {
				factor = 2
				has_government = democratic
			}
			modifier = {
				factor = 1
				has_government = neutrality
			}
			modifier = {
				factor = 0
				has_government = fascism
			}
		}
	}
	
	option = { # Steal their ideas
		name = deluge_steal_their_ideas
				
		add_political_power = -150
		
		add_popularity = {
			ideology = communism
			popularity = -0.1
		}
		
		add_ideas = welfare_state
		
		ai_chance = {
			factor = 1
			
			modifier = {
				factor = 1
				has_government = democratic
			}
			modifier = {
				factor = 1
				has_government = neutrality
			}
			modifier = {
				factor = 1
				has_government = fascism
			}
		}
	}
	
	option = { # Cancel the election
		name = deluge_cancel_the_election
				
		add_political_power = -150

		add_popularity = {
			ideology = communism
			popularity = 0.1
		}
		
		add_ideas = novote
		
		if = {
			limit = { has_government = democratic }
			set_politics = {
				ruling_party = democratic
				elections_allowed = no
			}
		}
		
		if = {
			limit = { has_government = neutrality }
			set_politics = {
				ruling_party = neutrality
				elections_allowed = no
			}
		}
		
		if = {
			limit = { has_government = fascism }
			set_politics = {
				ruling_party = fascism
				elections_allowed = no
			}
		}
		
		ai_chance = {
			factor = 1
			
			modifier = {
				factor = 0
				has_government = democratic
			}
			modifier = {
				factor = 1
				has_government = neutrality
			}
			modifier = {
				factor = 2
				has_government = fascism
			}
		}
	}
}
