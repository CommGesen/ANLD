					id = deluge_spain.10
					hours = 10
				}
			}
			hidden_effect = {
				news_event = {
					id = deluge_spain.17
					hours = 5
				}
			}
		}
		if = {
			limit = {
				country_exists = POR
				POR = {
					is_subject = no
				}
				POR = {
					has_government = neutrality
				}
			}
			POR = {
				country_event = {
					id = deluge_spain.7
				}
			}
		}
		if = {
			limit = {
				NOT = {
					SPR = {
						is_subject_of = FRA
					}
				}
			}

			factor = 1
		}
