# Keulen -> Köln
# Rijnland -> Westrhein

country_event = {
	id = deluge_names.1
	title = deluge_names1title
	desc = deluge_names1desc
	picture = GFX_report_event_generic_read_write
	
	trigger = {
		OR = {
			tag = GER
			tag = DDR
			tag = WGR
			tag = AUS
		}
		controls_state = 784
		NOT = {
			has_global_flag = name_westrhein
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	immediate = {
		set_global_flag = name_westrhein
	}
	
	option = {
		name = deluge_verygood

		set_province_name = { id = 587 name = "Köln" }
		784 = { set_state_name = "Westrhein" }
	}
}
